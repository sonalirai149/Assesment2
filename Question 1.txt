import java.util.*;
import java.io.*;
import java.util.Scanner;
public class UserMainCode {
  public static int calculateElectricityBill (String pr, String cur, int per_unit) {
    int pr1 = Integer.parseInt(pr.substring(5, pr.length()));
    int cur1 = Integer.parseInt(cur.substring(5, cur.length()));
    int output = Math.abs((cur1 - pr1) * per_unit);
    return output;
  }

    public static void main(String[] args) {
    Scanner sc = new Scanner(System.in);
    String inp1 = sc.nextLine();
    String inp2 = sc.nextLine();
    int inp3= sc.nextInt();
    int out = calculateElectricityBill (inp1, inp2, inp3);
    System.out.println(out);
    sc.close();
  }
}
